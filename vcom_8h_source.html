182static inline int
183_stk_excp(struct stk_vc *stk, char * msg)
184{
185 static time_t pre_excp_t = 0;
186 time_t now_excp_t;
187
188 if(stk_bot(stk)){
189 printf("%s : at the bottom of stack now\n", __func__);
190 return -1;
191 }
192
193 printf("stack exception !! %s\n", msg);
194 stk->top = 0;
195 mon_update_check(stk, 1, msg);
196
197 if(pre_excp_t == 0){ //mean first time excption
198 time(&pre_excp_t);
199 }else{
200 time(&now_excp_t);
201 if((now_excp_t-pre_excp_t) < EXCP_RECONN_TIME){
202 sleep(EXCP_SLEEPTIME);
203 }
204 pre_excp_t = now_excp_t;
205 }
206
207 return 0;
208}

326static inline int fdcheck(int fd, int type, struct timeval * ctv)
327{
328 fd_set rfds;
329 fd_set wfds;
330 struct timeval tv;
331 int ret;
332
333 tv.tv_sec = 0;
334 tv.tv_usec = 0;
335
336 FD_ZERO(&rfds);
337 FD_ZERO(&wfds);
338 if(type& FD_RD_RDY){
339 FD_SET(fd, &rfds);
340 }
341 if(type& FD_WR_RDY){
342 FD_SET(fd, &wfds);
343 }
344 if(ctv > 0){
345 ret = select(fd + 1, &rfds, &wfds, 0, ctv);
346 }else{
347 ret = select(fd + 1, &rfds, &wfds, 0, &tv);
348 }
349
350 if(ret == 0)
351 return 0;
352
353 ret = 0;
354
355 if(FD_ISSET(fd, &rfds)){
356 ret |= FD_RD_RDY;
357 }
358
359 if(FD_ISSET(fd, &wfds)){
360 ret |= FD_WR_RDY;
361 }
362
363 return ret;
364}
